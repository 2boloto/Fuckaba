// Получает из сети байты до конца строки включительно, все они игнорируются. Работает только строками структур ХТТП, то есть кончающимися на `\r\n`.

// См. `start-receiving.txt`.

// >- r
// >  6 нулей
// >  1, если надо проверить, не пустая ли строка, иначе 0

// <- новый r, если хорошо, иначе 0
// <  6 нулей
// <  1, если на входе была 1 и строка оказалась пустая, иначе 0

// В случае ошибки этот флаг может остаться единицей.

#INCREASE 3 <>4

#INCREASE 2 <>3
#WHILE <>3
	#INCLUDE "receive-byte.txt"

	#IF_SAVE_FLAG
		#VISIT 5
			#INPUT

			#INCREASE -10
			#SUB_TO 1 <>-1

			#IF
				#ZERO <>-2
				#INCREASE 2 <>-2
				#INCREASE 3 <>-1
				#INCREASE 1 <>1
				#ZERO <>2
			#END

			#BOOL_UNLESS <>1
				#INCREASE -1 <>-2
			#END
		#END
	#END

	#ELSE
		#ZERO <>3
	#END
#END

#ZERO <>4
