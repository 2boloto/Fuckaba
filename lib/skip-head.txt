// Получает из сети байты до конца блока заголовков, включая последнюю пустую строку, все эти байты игнорируются.

// См. `start-receiving.txt`.

// >- r
// >  8 нулей

// <- новый r, если хорошо, иначе 0
// <  8 нулей

#ADD_TO 1

#INCREASE 1
#WHILE
	#VISIT 1
		#INCREASE 1 <>7
		#INCLUDE "skip-line.txt"

		#SUB_TO -8 <>7

		#UNLESS_SAVE
			#ZERO <>-1
		#END
	#END
#END

#ADD_TO -1 <>1
